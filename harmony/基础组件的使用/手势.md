
手势滑动的监听
```
 panOption = new PanGestureOptions({ direction: PanDirection.Left | PanDirection.Right })
  
 aboutToAppear(): void {
    //this.panOption.setDirection(PanDirection.All) //以任意方向拖动
    //this.panOption.setFingers(2)  //指定两个手指
 }
 Column(){
    
    }
    .width('100%')
    .height('100%')
    .backgroundColor(Color.Red)
    .gesture(
      PanGesture(this.panOption)
        .onActionStart(()=>{
          LogUtil.debug('手势：开始')
        })
        .onActionUpdate(()=>{
          LogUtil.debug('手势：更新')
        })
        .onActionEnd(()=>{
          LogUtil.debug('手势：结束')
        })
    )
  }
```
