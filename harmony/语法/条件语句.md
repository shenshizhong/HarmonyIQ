```
通过if else 进行条件判断：
if num: number =5
if(num > 0){

通过switch case 进行条件判断：
var grade: String = 'A';
switch(grade){
    case 'A':{
        console.log('优');
        break;
    }
     case 'B':{
        console.log('良');
        break;
    }
     case 'C':{
        console.log('及格');
        break;
    }
     case 'D':{
        console.log('不及格');
        break;
    }
     default:{
        console.log('非法输入');
        break;
    }
}
```
